```java
// 第181-227行
@Scheduled(fixedRate = 30000)  // 每30秒
public void checkSoldOutActivities() {
    // 第183-189行：查询进行中的活动
    List<SeckillActivity> ongoingActivities = seckillActivityMapper.selectList(...);
    
    for (SeckillActivity activity : ongoingActivities) {
        // 第194-197行：检查DB库存
        if (activity.getRemainStock() > 0) {
            continue;  // 还有库存，跳过
        }
        
        // 第199-206行：检查待处理订单
        long pendingCount = seckillOrderMapper.selectCount(...);
        if (pendingCount > 0) {
            continue;  // 有待处理订单，可能会回滚
        }
        
        // 第209-220行：标记为已售罄
        seckillActivityMapper.update(...);  // 标记SOLD_OUT
        redisTemplate.delete(stockKey);     // 清理Redis缓存
        redisTemplate.delete(activityKey);
        log.info("秒杀活动提前售罄...");
    }
}
```

```java
    // 第171-176行：遍历处理

    }
}
```
